String Comparison

Functions for comparing strings:

#include <stdio.h>
#include <string.h>

int main() {
    // Example of string comparison
    char str1[] = "apple";
    char str2[] = "banana";
    int result = strcmp(str1, str2);
    if (result < 0) {
        printf("%s is less than %s\n", str1, str2);
    } else if (result > 0) {
        printf("%s is greater than %s\n", str1, str2);
    } else {
        printf("%s is equal to %s\n", str1, str2);
    }

    return 0;
}

This snippet uses `strcmp` to compare two strings.

Searching in Strings

Functions for searching within strings:

#include <stdio.h>
#include <string.h>

int main() {
    // Example of string searching
    char str[] = "The quick brown fox";
    char substr[] = "brown";
    char *ptr = strstr(str, substr);
    if (ptr != NULL) {
        printf("'%s' found at position %ld in '%s'\n", substr, ptr - str, str);
    } else {
        printf("'%s' not found in '%s'\n", substr, str);
    }

    return 0;
}

This code uses `strstr` to search for a substring within a string.
